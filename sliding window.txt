import RPi.GPIO as GPIO
import time

# Define servo pin
servo_pin = 18  # GPIO pin 18

# Set up GPIO
GPIO.setmode(GPIO.BCM)
GPIO.setup(servo_pin, GPIO.OUT)

# Create PWM instance for the servo
servo = GPIO.PWM(servo_pin, 50)  # 50 Hz frequency

# Start PWM with 0% duty cycle (servo at 0 degrees)
servo.start(0)

def move_servo_smoothly(angle):
    # Convert angle (0 to 180) to the appropriate duty cycle
    duty_cycle = (angle / 18.0) + 2.5
    servo.ChangeDutyCycle(duty_cycle)
    time.sleep(1)  # Adjust the sleep time as needed

# Move the servo smoothly from 0 to 90 to 180 degrees
try:
    while True:
        move_servo_smoothly(0)  # Move to 0 degrees
        print("Servo at 0 degrees")
        time.sleep(1)
        
        move_servo_smoothly(90)  # Move to 90 degrees
        print("Servo at 90 degrees")
        time.sleep(1)

        move_servo_smoothly(180)  # Move to 180 degrees
        print("Servo at 180 degrees")
        time.sleep(1)

except KeyboardInterrupt:
    pass

finally:
    # Clean up GPIO on exit
    servo.stop()
    GPIO.cleanup()
